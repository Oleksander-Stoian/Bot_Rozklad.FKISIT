===========================================================
             СТРУКТУРА ПРОЄКТУ
===========================================================

📂 FinalProject/              <-- Коренева папка проєкту
│
├── 📜 main.py                <-- ТОЧКА ВХОДУ. Запускає бота, реєструє хендлери та планувальник.
├── 📜 config.py              <-- КОНФІГУРАЦІЯ. Завантажує змінні середовища та налаштування розкладу дзвінків.
├── 📜 .env                   <-- СЕКРЕТИ. Містить токен бота та налаштування підключення до Redis.
├── 📜 rozklad_pro.xlsx       <-- БАЗА ДАНИХ РОЗКЛАДУ. Excel-файл, з якого бот читає пари.
├── 📜 requirements.txt       <-- ЗАЛЕЖНОСТІ. Список бібліотек для встановлення (aiogram, pandas, redis).
├── 📜 start.bat              <-- ЛЕГКИЙ ЗАПУСК. Скрипт для автоматичного встановлення бібліотек та старту.
│
├── 📂 core/                  <-- ЯДРО БОТА
│   ├── bot.py                <-- Ініціалізація об'єкта Bot та Dispatcher.
│   ├── scheduler.py          <-- ПЛАНУВАЛЬНИК. Перевіряє час щохвилини та надсилає сповіщення про пари.
│   └── states.py             <-- FSM. Визначення станів (States) для машинних станів (діалогів).
│
├── 📂 handlers/              <-- ОБРОБНИКИ (Логіка відповідей на команди)
│   ├── start.py              <-- Обробка команди /start та вибір ролі.
│   ├── student.py            <-- Логіка для студента: вибір курсу, групи, збереження налаштувань.
│   ├── teacher.py            <-- Логіка для викладача: пошук себе у списку, збереження.
│   ├── schedule.py           <-- Вивід розкладу: "Яка зараз пара", "Сьогодні", "Тиждень".
│   └── bells.py              <-- Вивід розкладу дзвінків.
│
├── 📂 keyboards/             <-- ІНТЕРФЕЙС (Кнопки)
│   ├── builders.py           <-- Inline-клавіатури (кнопки під повідомленнями), що будуються динамічно.
│   └── reply.py              <-- Головне меню (кнопки знизу екрану).
│
├── 📂 services/              <-- БІЗНЕС-ЛОГІКА (Робота з даними)
│   ├── excel_service.py      <-- Робота з Pandas: читання Excel, фільтрація розкладу, кешування.
│   ├── redis_service.py      <-- Робота з БД: збереження ролей, груп та налаштувань користувачів.
│   ├── week_service.py       <-- Логіка визначення навчального тижня (Чисельник/Знаменник).
│   └── bell_service.py       <-- Перевірка, чи триває зараз урок (для функції "Яка зараз пара").
│
└── 📂 utils/                 <-- ДОПОМІЖНІ УТИЛІТИ
    └── formatters.py         <-- Форматування тексту: красивий вивід "мигалок", додавання емодзі.

===========================================================
